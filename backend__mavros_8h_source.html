46 class HistoryBuffer { // TODO: template? utils?
47 public:
48  void set_size(size_t _size) { buffer_size_ = _size; }
49 
50  void reset() {
51  std::lock_guard<std::mutex> lock(mutex_);
52  buffer_.clear();
53  }
54 
55  void update(double _value) {
56  std::lock_guard<std::mutex> lock(mutex_);
57  buffer_.push_back(_value);
58  if (buffer_.size() > buffer_size_) {
59  buffer_.pop_front();
60  }
61  }
62 
63  bool metrics(double& _min, double& _mean, double& _max) {
64  std::lock_guard<std::mutex> lock(mutex_);
65  if (buffer_.size() >= buffer_size_) {
66  double min_value = +std::numeric_limits<double>::max();
67  double max_value = -std::numeric_limits<double>::max();
68  double sum = 0;
69  for (int i = 0; i < buffer_.size(); i++) {
70  if (buffer_[i] < min_value) { min_value = buffer_[i]; }
71  if (buffer_[i] > max_value) { max_value = buffer_[i]; }
72  sum += buffer_[i];
73  }
74  _min = min_value;
75  _max = max_value;
76  _mean = sum / buffer_.size();
77  return true;
78  }
79  return false;
80  }
81 
82 protected:
83  size_t buffer_size_ = 0;
84  std::deque<double> buffer_;
85  std::mutex mutex_;
86 };
